139  SlewType compute_slew(lemon::ListDigraph::Node node, CapacitanceType load) const {
140  SlewType worst_slew = MergeStrategy::best();
141  if(lemon::countInArcs(m_topology.g.G(), node) == 0) // PI without driver
142  return m_timing.nodes.slew(node);
143  switch(m_topology.g.node_edge(node))
144  {
145  case edges::RISE:
146  for(lemon::ListDigraph::InArcIt it(m_topology.g.G(), node); it != lemon::INVALID; ++it)
147  {
148  auto tarc = m_topology.g.edge_entity(it) ;
149  worst_slew = m_merge(worst_slew, m_timing.lib.timing_arc_rise_slew(tarc).compute(load, m_timing.nodes.slew(m_topology.g.edge_source(it))));
150  }
151  break;
152  case edges::FALL:
153  for(lemon::ListDigraph::InArcIt it(m_topology.g.G(), node); it != lemon::INVALID; ++it)
154  {
155  auto tarc = m_topology.g.edge_entity(it) ;
156  worst_slew = m_merge(worst_slew, m_timing.lib.timing_arc_fall_slew(tarc).compute(load, m_timing.nodes.slew(m_topology.g.edge_source(it))));
157  }
158  break;
159  }
160  return worst_slew;
161  }

239  void update_ats() {
240  {
241  for(auto & level : m_topology.levels)
242  {
243  std::size_t i;
244  for(i = 0; i < level.size(); ++i) // paralell
245  {
246  auto node = level[i];
247  if(lemon::countInArcs(m_topology.g.G(), node) != 0)
248  {
249  auto pin = m_topology.g.pin(node);
250  auto net = m_topology.netlist.pin_net(pin);
251  auto tree = m_rc_trees[m_topology.netlist.net_system().lookup(net)];
252  lemon::ListGraph::NodeMap< SlewType > slews(tree.graph());
253  lemon::ListGraph::NodeMap< SlewType > delays(tree.graph());
254  lemon::ListGraph::NodeMap< CapacitanceType > ceffs(tree.graph());
255  WireDelayModel calculator;
256  calculator.delay_map(delays);
257  calculator.slew_map(slews);
258  calculator.ceff_map(ceffs);
259  std::function<SlewType(CapacitanceType)> s_calculator = std::bind(&generic_sta::compute_slew, this, node, std::placeholders::_1);
260  auto source_capacitor = tree.capacitor_by_name(m_topology.netlist.pin_name(pin));
261  CapacitanceType load = calculator.simulate(s_calculator, tree, source_capacitor);
262 
263  m_timing.nodes.load(node, load);
264  m_timing.nodes.slew(node, slews[source_capacitor]);
265 
266  SlewType worst_arrival = MergeStrategy::best();
267  switch(m_topology.g.node_edge(node))
268  {
269  case edges::RISE:
270  for(lemon::ListDigraph::InArcIt it(m_topology.g.G(), node); it != lemon::INVALID; ++it)
271  {
272  auto tarc = m_topology.g.edge_entity(it) ;
273  auto edge_source = m_topology.g.edge_source(it);
274  auto arc_delay = m_timing.lib.timing_arc_rise_delay(tarc).compute(load, m_timing.nodes.slew(edge_source));
275  auto arc_slew = m_timing.lib.timing_arc_rise_slew(tarc).compute(load, m_timing.nodes.slew(edge_source));
276  m_timing.arcs.delay(it, arc_delay);
277  m_timing.arcs.slew(it, arc_slew);
278  worst_arrival = m_merge(worst_arrival, m_timing.nodes.arrival(edge_source) + arc_delay);
279  }
280  break;
281  case edges::FALL:
282  for(lemon::ListDigraph::InArcIt it(m_topology.g.G(), node); it != lemon::INVALID; ++it)
283  {
284  auto tarc = m_topology.g.edge_entity(it) ;
285  auto edge_source = m_topology.g.edge_source(it);
286  auto arc_delay = m_timing.lib.timing_arc_fall_delay(tarc).compute(load, m_timing.nodes.slew(edge_source));
287  auto arc_slew = m_timing.lib.timing_arc_fall_slew(tarc).compute(load, m_timing.nodes.slew(edge_source));
288  m_timing.arcs.delay(it, arc_delay);
289  m_timing.arcs.slew(it, arc_slew);
290  worst_arrival = m_merge(worst_arrival, m_timing.nodes.arrival(edge_source) + arc_delay);
291  }
292  break;
293  }
294  m_timing.nodes.arrival(node, worst_arrival);
295 
296  for(lemon::ListDigraph::OutArcIt arc(m_topology.g.G(), node); arc != lemon::INVALID; ++arc)
297  {
298  auto arc_target = m_topology.g.edge_target(arc);
299  auto target_pin = m_topology.g.pin(arc_target);
300  auto target_capacitor = tree.capacitor_by_name(m_topology.netlist.pin_name(target_pin));
301  m_timing.arcs.slew(arc, slews[target_capacitor]);
302  m_timing.arcs.delay(arc, delays[target_capacitor]);
303  m_timing.nodes.slew(arc_target, m_timing.arcs.slew(arc));
304  m_timing.nodes.arrival(arc_target, m_timing.nodes.arrival(node) + m_timing.arcs.delay(arc));
305  }
306  }
307  }
308  }
309 
310  }
