76 
77  lemon::ListGraph::NodeMap< SlewType > & slews = *m_slews;
78  lemon::ListGraph::NodeMap< SlewType > & delays = *m_delays;
79  lemon::ListGraph::NodeMap< CapacitanceType > & ceff = *m_ceff;
80 
81  elmore delay(tree, source);
82  elmore_second_moment second_moment(tree, delay);
83  auto source_slew = slew_calculator(tree.lumped());
84 
85  for(lemon::ListGraph::NodeIt node(tree.graph()); node != lemon::INVALID; ++node)
86  {
87  delays[node] = delay.at(node);
88  auto step_slew = boost::units::sqrt( second_moment.at(node)*2.0 - boost::units::pow<2>(delay.at(node)) );
89  slews[node] = boost::units::sqrt(boost::units::pow<2>(source_slew) + boost::units::pow<2>(step_slew));
90  }
91 
92  return tree.lumped();
93  }

98 {

100  using SlewType = boost::units::quantity < boost::units::si::time >;

129  template <class SlewCalculator>
130  CapacitanceType simulate(const SlewCalculator & slew_calculator, const interconnection::rc_tree & tree, const interconnection::rc_tree::capacitor_id source)
131  {
132  if(!m_slews)
133  m_slews = new lemon::ListGraph::NodeMap< SlewType >(tree.graph());
134  if(!m_delays)
135  m_delays = new lemon::ListGraph::NodeMap< SlewType >(tree.graph());
136  if(!m_ceff)
137  m_ceff = new lemon::ListGraph::NodeMap< CapacitanceType >(tree.graph());
138 
139  lemon::ListGraph::NodeMap< SlewType > & slews = *m_slews;
140  lemon::ListGraph::NodeMap< SlewType > & delays = *m_delays;
141  lemon::ListGraph::NodeMap< CapacitanceType > & ceff = *m_ceff;
142 
143  lemon::ListGraph::NodeMap< lemon::ListGraph::Arc > pred(tree.graph());
144  for(lemon::ListGraph::NodeIt it(tree.graph()); it != lemon::INVALID; ++it)
145  pred[it] = lemon::INVALID;
146 
147  std::deque<lemon::ListGraph::Node> to_process;
148  std::vector< lemon::ListGraph::Node > order(tree.capacitor_count());
149  order.resize(0);
150  to_process.push_back(source);
151  while(!to_process.empty())
152  {
153  auto current = to_process.front();
154  to_process.pop_front();
155  order.push_back(current);
156  for(interconnection::rc_tree::resistor_it it(tree.capacitor_resistors(current)); it != lemon::INVALID; ++it)
157  {
158  auto opposite_capacitor = tree.other_capacitor(it, current);
159  if(opposite_capacitor != source && pred[opposite_capacitor] == lemon::INVALID)
160  {
161  pred[opposite_capacitor] = it;
162  to_process.push_back(opposite_capacitor);
163  }
164  }
165  }
166  assert(pred[source] == lemon::INVALID);
167  for(std::size_t i{0}; i < order.size(); ++i)
168  {
169  auto current = order[i];
170  ceff[current] = tree.capacitance(current);
171  }
172  for(std::size_t i = order.size()-1; i > 0; --i)
173  {
174  auto current = order[i];
175  auto parent = tree.graph().oppositeNode(current, pred[current]);
176  ceff[parent] += ceff[current];
177  }
178 
179 
180  double error = 1.0;
181  std::size_t iteration = 0;
182 
183  CapacitanceType current_ceff;
184  delays[source] = SlewType(0.0*boost::units::si::seconds);
185  while (error > m_precision) {
186  current_ceff = ceff[source];
187 // std::cout << "iteration #" << iteration++ << " ceff " << current_ceff << std::endl;
188  slews[source] = slew_calculator(current_ceff);
189  for(std::size_t i = 1; i < order.size(); ++i)
190  {
191  auto current = order[i];
192  auto parent = tree.graph().oppositeNode(order[i], pred[order[i]]);
193  auto resistance_with_parent = tree.resistance(pred[order[i]]);
194 
195  slews[current] = slews[parent];
196  if(slews[parent] > 0.0*boost::units::si::seconds){
197  double x = resistance_with_parent*ceff[current]/slews[parent];
198  slews[current] = slews[parent]/ (1-x*(1-std::exp(-1/x)));
199  }
200  delays[current] = delays[parent] + resistance_with_parent*ceff[current];
201  }
202  for(auto node : order)
203  ceff[node] = tree.capacitance(node);
204  for(auto node_it = order.rbegin(); node_it != order.rend(); ++node_it)
205  {
206  auto current = *node_it;
207  if(pred[*node_it] != lemon::INVALID)
208  {
209  auto parent = tree.graph().oppositeNode(current, pred[current]);
210  auto resistance_with_parent = tree.resistance(pred[current]);
211  double x = 2.0 * resistance_with_parent * ceff[current] / slews[parent];
212  double y = 1.0 - std::exp(-1.0/x);
213  double shielding_factor = (slews[parent] > 0.0*boost::units::si::seconds?1.0 - x * y:1.0);
214  ceff[parent] += shielding_factor*ceff[current];
215 
216  }
217  }
218  error = boost::units::abs(current_ceff-ceff[source])/std::max(current_ceff, ceff[source]);
219  }
220  return current_ceff;
221  }
222 };
